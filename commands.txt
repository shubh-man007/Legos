## Database:
# Info:
gcloud sql instances list 
gcloud sql instances describe legos-user
gcloud sql instances describe legos-user --format="value(ipAddresses[0].ipAddress)"
gcloud sql instances describe legos-user --format="value(connectionName)"

# Create database:
gcloud sql databases create legos_ai --instance=legos-user  

# Create a User:
gcloud sql users create legos_user --instance=legos-user --password=password
